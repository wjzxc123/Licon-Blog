#### 请求基础类
```java
@Getter
@Setter
public abstract class BaseRequest implements Serializable {

    @Serial
    private static final long serialVersionUID = 3182217732056533200L;

    private String traceId;

    private String clientVersion;

    private Long timestamp;


    @NotNull(message = "traceId 不能为空")
    @Pattern(regexp = "^[a-zA-Z0-9]{32}$", message = "traceId 格式错误")
    public String traceId() {
        return traceId;
    }


    public abstract void validate();

}

```

#### 分页请求参数基础类
```java
@Getter
@Setter
public abstract class PageRequest extends BaseRequest{

    @Min(value = 1, message = "page 不能小于 1")
    private int page;

    @Max(value = 100, message = "pageSize 不能大于 100")
    private int pageSize;

    private String sortFiled;

    private SortType sortType;


    @Override
    public void validate() {

    }
}

```

排序枚举
```java
public enum SortType {

    ASC,

    DESC;
}
```


#### 分页参数包装类
```java
@Getter
@Setter
@ToString
public class PageParam<T> extends PageRequest{

    private T data;

}

```

#### 业务请求参数
```java
@Getter
@Setter
@ToString
public class LiconRequestParam extends BaseRequest {

    private String username;

    private String password;

    @Override
    public void validate() {
        if (!StringUtils.hasText(username)){
            throw new RuntimeException("非法的用户名");
        }

        if (!StringUtils.hasText(username)){
            throw new RuntimeException("非法的密码");
        }
    }
}
```
